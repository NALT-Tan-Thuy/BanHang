Phương thức Miêu tả Ví dụ
concat()

Trả về một mảng mới bao gồm mảng này kết hợp với các mảng khác và/hoặc giá trị khác.

arr1.concat(arr2, arr3, ..., arrX) 

var array1 = ['a', 'b', 'c'];
var array2 = ['d', 'e', 'f'];

console.log(array1.concat(array2));
//=> Array ["a", "b", "c", "d", "e", "f"]
copyWithin() Ghi đè các phần tử nó sao chép được lên các phần tử của mảng tính từ một vị trí xác định bởi người dùng.

array.copyWithin(target, start, end)
+ target là vị trí mà hàm sẽ bắt đầu ghi đè các phần tử sao chép được.
+ start là vị trí bắt đầu sao chép.
+ end là vị trí kết thúc sao chép. 

var array1 = [1, 2, 3, 4, 5];

// place at position 0 the element between
// position 3 and 4
console.log(array1.copyWithin(0, 3, 4));
// output: Array [4, 2, 3, 4, 5]

// place at position 1 the elements after position 3
console.log(array1.copyWithin(1, 3));
// output: Array [4, 4, 5, 4, 5]
fill()

Thay thế các phần tử trong một mảng từ vị trí start đến vị trí end với một giá trị value

arr.fill(value, start, end)

 
var array1 = [1, 2, 3, 4];

// fill with 0 from position 2 until position 4
console.log(array1.fill(0, 2, 4));
// output: [1, 2, 0, 0]

// fill with 5 from position 1
console.log(array1.fill(5, 1));
// output: [1, 5, 5, 5]

console.log(array1.fill(6));
// output: [6, 6, 6, 6]
filter()

Tạo một mảng mới với tất cả các phần tử của mảng này, mà hàm lọc đã cho trả về true.

 
var words = [1, 2, 3, 4, 5, 6];

const result = words.filter(num => num > 3);

console.log(result);
// output: Array [4, 5, 6]

function isBigEnough(value){
	return value > 10;  
}
var fil = [12, 5, 8, 130, 44].filter(isBigEnough);
console.log(fil);
// fil is [12, 130, 44]
find()

Trả về giá trị của phần tử đầu tiên trong mảng thỏa mãn được điều kiện kiểm tra

 
var inventory = [
    {name: 'apples', quantity: 2},
    {name: 'bananas', quantity: 0},
    {name: 'cherries', quantity: 5}
];

function isCherries(fruit) { 
    return fruit.name === 'cherries';
}

console.log(inventory.find(isCherries)); 
// Object { name: "cherries", quantity: 5 }

const result = inventory.find(fruit =>
fruit.name === 'cherries' );
console.log(result) 
// Object { name: "cherries", quantity: 5 }
findIndex()

Trả về chỉ mục đầu tiên (thấp nhất) của một phần tử trong mảng tương đương với giá trị đã cho, hoặc -1 nếu không được tìm thấy.

 
var array1 = [5, 12, 8, 130, 44];
function findFirstLargeNumber(element) {
    return element > 13;
  }
  console.log(array1.findIndex(findFirstLargeNumber));
  // expected output: 3
  
  function isPrime(element, index, array) {
    var start = 2;
    while (start <= Math.sqrt(element)) {
      if (element % start < 1) {
        return false;
      } else start++;
    }
    return element > 1;
  }
  
  console.log([4, 6, 8, 12].findIndex(isPrime)); 
  // -1, not found
  console.log([4, 6, 7, 12].findIndex(isPrime)); 
  // 2
forEach()

Truyền lần lượt các phần tử của mảng vào một hàm được cung câp trước đó, hàm này sẽ tiến hành xử lý các thành phần tử cửa mảng như là một tham số truyền vào

 
const items = ['item1', 'item2', 'item3'];
const copy = [];

items.forEach(function(item){
  copy.push(item)
});

console.log(copy);
// Array ["item1", "item2", "item3"]
from()

Trả về đối tượng Array từ bất kỳ đối tượng nào có thuộc tính length hoặc đối tượng có thể lặp lại.(Điển hình: trả về một mảng char từ 1 string)

Array.from(object, mapFunction, thisValue)

 
console.log(Array.from('foo'));
// output: Array ["f", "o", "o"]

console.log(Array.from([1, 2, 3], x => x + x));
// output: Array [2, 4, 6]

console.log(Array.from({length: 5}, (v, i) => i));
// output: Array [0, 1, 2, 3, 4]

Kiểm tra xem một mảng có chứa phần tử element hay không

array.includes(element, start) 

var array1 = [1, 2, 3];
console.log(array1.includes(2));
// output: true

var pets = ['cat', 'dog', 'bat'];
console.log(pets.includes('cat'));
// output: true

console.log(pets.includes('at'));
// output: false
indexOf()

Trả về chỉ mục đầu tiên (thấp nhất) của một phần tử trong mảng tương đương với giá trị đã cho, hoặc -1 nếu không được tìm thấy.

arr.indexOf(item, start) 

var beasts = ['ant', 'bison', 'camel', 'duck', 'bison'];

console.log(beasts.indexOf('bison'));
// expected output: 1

// start from index 2
console.log(beasts.indexOf('bison', 2));
// expected output: 4

console.log(beasts.indexOf('giraffe'));
// expected output: -1

Kiểm tra một đối tượng có phải là một mảng hay không

Array.isArray(obj)

join()

Nối các phần tử của mảng thành một chuỗi, các phần tử được ngăn cách nhau bởi kí tự do người dùng quy định, mặc định là dấu ",".

arr.join(separator) 

var elements = ['Fire', 'Wind', 'Rain'];

console.log(elements.join());
// output: Fire,Wind,Rain

console.log(elements.join(''));
// output: FireWindRain

console.log(elements.join('-'));
// output: Fire-Wind-Rain

function f(a, b, c) {
  var s = Array.prototype.join.call(arguments);
  console.log(s); // '1,a,true'
}
f(1, 'a', true);
// output: "1,a,true"
lastIndexOf()

Trả về chỉ mục cuối cùng (lớn nhất) của một phần tử trong mảng tương đương với giá trị đã cho, hoặc -1 nếu không được tìm thấy.

arr.lastIndexOf(item, start) 

var numbers = [2, 5, 9, 2];
numbers.lastIndexOf(2);     // 3
numbers.lastIndexOf(7);     // -1
numbers.lastIndexOf(2, 3);  // 3
numbers.lastIndexOf(2, 2);  // 0
numbers.lastIndexOf(2, -2); // 0
numbers.lastIndexOf(2, -1); // 3
map()

Tạo một mảng mới với các kết quả của việc gọi một hàm đã cho trên mỗi phần tử của mảng này.

 
var array1 = [1, 4, 9, 16];
const map1 = array1.map(x => x * 2);
console.log(map1);
//output: Array [2, 8, 18, 32]

var kvArray = [{key: 1, value: 10}, 
               {key: 2, value: 20}, 
               {key: 3, value: 30}];
var reformattedArray = kvArray.map(obj =>{ 
   var rObj = {};
   rObj[obj.key] = obj.value;
   return rObj;
});
console.log(reformattedArray);
// output: Array [Object { 1: 10 }, 
// Object { 2: 20 }, Object { 3: 30 }]

reduce()

Áp dụng một hàm đồng thời với hai giá trị của mảng (từ trái qua phải) khi để giảm nó tới một giá trị đơn.

 
const array1 = [1, 2, 3, 4];
const reducer = (a, b) => a + b;

// 1 + 2 + 3 + 4
console.log(array1.reduce(reducer));
// output: 10

// 5 + 1 + 2 + 3 + 4
console.log(array1.reduce(reducer, 5));
// output: 15

Áp dụng một hàm đồng thời với hai giá trị của mảng (từ phải qua trái) khi để giảm nó tới một giá trị đơn.

 
const array1 = [[0, 1], [2, 3], [4, 5]].reduceRight(
  (previous, current) => previous.concat(current)
  );
  
  console.log(array1);
  // output: Array [4, 5, 2, 3, 0, 1]
  
  var arr = [0, 1, 2, 3, 4];
  var sum = arr.reduceRight(
    function(previousValue, currentValue, index, array) {
    return previousValue + currentValue;
  });
  console.log(sum);
  // output: 10
reverse()

Đảo ngược thứ tự của các phần tử của một mảng – Đầu tiên trở thành cuối cùng và cuối cùng trở thành đầu tiên.

arr.reverse() 

var array1 = [1, 2, 3];
console.log('array1: ', array1);
// output: "array1: " Array [1, 2, 3]

var reversed = array1.reverse(); 
console.log('reversed: ', reversed);
// expected output: "reversed: " Array [3, 2, 1]

console.log('array1: ', array1);
// output: "array1: " Array [3, 2, 1]

slice()

Extract – trích một đoạn của một mảng từ vị trí begin đến cuối (hoặc end) và trả về một mảng mới.

arr.slice( begin [,end] ); 

var arr = [1, 2, 3, 4, 5];

console.log(arr.slice(2));

console.log(arr.slice(2, 4));

some()

Trả về true nếu có ít nhất một phần tử trong mảng này thỏa mãn hàm kiểm tra đã cho.

arr.some(callback[, thisArg]) 

function isBiggerThan10(element, index, array) {
  return element > 10;
  }
  
  [2, 5, 8, 1, 4].some(isBiggerThan10);  // false
  [12, 5, 8, 1, 4].some(isBiggerThan10); // true
  
  [2, 5, 8, 1, 4].some(x => x > 10);  // false
  [12, 5, 8, 1, 4].some(x => x > 10); // tru
sort()

sắp xếp các phần tử trong một mảng, các phần tử có thể được sắp xếp theo bảng chữ cái hoặc theo chữ số theo thứ tự tăng dần hoặc giảm dần

array.sort(compareFunction) 

var months = ['March', 'Jan', 'Feb', 'Dec'];
months.sort();
console.log(months);
// output: Array ["Dec", "Feb", "Jan", "March"]

var array1 = [1, 30, 4, 21];
array1.sort();
console.log(array1);
//output: Array [1, 21, 30, 4]

array1.sort((a, b) => a - b);
console.log(array1);
//output: Array [1, 4, 21, 30]

array1.sort((a, b) => b - a);
console.log(array1);
// output: Array [30, 21, 4, 1]
splice()

Thêm và/hoặc gỡ bỏ các phần tử từ một mảng.

Nếu howMany = 0: Không xóa phần tử nào, chỉ có thêm các phần tử từ vị trí index.

Nếu howMany = a > 0: Xóa đi a phần tử từ vị trí index rồi thêm các phần tử vào mảng

arr.splice(index, howMany, [element1][, ..., elementN]); 

var months = ['Jan', 'March', 'April', 'June'];
months.splice(1, 0, 'Feb');
// inserts at 1st index position
console.log(months);
// output: Array ['Jan', 'Feb', 'March', 'April', 'June']

months.splice(4, 1, 'May');
// replaces 1 element at 4th index
console.log(months);
// output: Array ['Jan', 'Feb', 'March', 'April', 'May']

unshift()

Thêm một hoặc nhiều phần tử tới phần đầu của một mảng và trả về độ dài mới của mảng.

arr.unshift(element1[, ...[, elementN]]) 

var array1 = [1, 2, 3];

console.log(array1.unshift(4, 5));
// output: 5
